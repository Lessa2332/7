            <img id="our-png" src="https://raw.githubusercontent.com/твій_юзер/твій_репозиторій/main/моя/overlay.png">
            <audio id="our-audio" src="https://raw.githubusercontent.com/твій_юзер/твій_репозиторій/main/моя/sound.mp3" preload="auto"></audio>
        </a-assets>
        <a-nft type="nft" url="https://raw.githubusercontent.com/твій_юзер/твій_репозиторій/main/моя/marker"
               smooth="true" smoothCount="10" smoothTolerance="0.01" smoothThreshold="5">

                     animation="property: rotation; to: 0 360 0; loop: true; dur: 5000"></a-image>

    <script>
        const overlayText = document.getElementById('overlay-text');
        const audio = document.getElementById('our-audio');
        const listenButton = document.getElementById('listen-button');
        const exitButton = document.getElementById('exit-button');

        // Обробка подій маркера
        AFRAME.registerComponent('markerhandler', {
            init: function () {
                this.el.addEventListener('markerFound', () => {
                    overlayText.style.display = 'block';
                    listenButton.style.display = 'block';
                    exitButton.style.display = 'block';
                });
                this.el.addEventListener('markerLost', () => {
                    overlayText.style.display = 'none';
                    audio.pause();
                    audio.currentTime = 0;
                });
            }
        });
        document.querySelector('a-nft').setAttribute('markerhandler', '');

        // Кнопка "Слухати"
        listenButton.addEventListener('click', () => {
            if (audio.paused) {
                audio.play();
                listenButton.textContent = 'Пауза';
            } else {
                audio.pause();
                listenButton.textContent = 'Слухати';
            }
        });

        // Кнопка "Вийти"
        exitButton.addEventListener('click', () => {
            audio.pause();
            audio.currentTime = 0;
            overlayText.style.display = 'none';
            listenButton.style.display = 'none';
            exitButton.style.display = 'none';
            window.location.reload(); // Перезавантажуємо сторінку для зупинки AR
        });

        // Адаптація розміру сцени при зміні розміру екрану
        window.addEventListener('resize', () => {
            const scene = document.querySelector('a-scene');
            scene.setAttribute('arjs', `sourceWidth: ${window.innerWidth}; sourceHeight: ${window.innerHeight}`);
        });
    </script>
